REST API con Spring Boot

Guida completa alla creazione di API RESTful con Spring Boot: CRUD, validazione, paginazione, exception handling e best practices

Cos'è una REST API?

REST (Representational State Transfer) è uno stile architetturale per la progettazione di API web. Le API RESTful utilizzano i metodi HTTP standard per eseguire operazioni CRUD su risorse.
Vantaggi delle API REST:
- Stateless: ogni richiesta contiene tutte le informazioni necessarie
- Cacheable: le risposte possono essere memorizzate nella cache
- Layered system: l'architettura può essere suddivisa in livelli
- Uniform interface: interfaccia standardizzata tra componenti

Codici di stato HTTP comuni:
- 200 OK: richiesta completata con successo
- 201 Created: risorsa creata con successo
- 204 No Content: richiesta completata, nessun contenuto da restituire
- 400 Bad Request: richiesta malformata
- 401 Unauthorized: autenticazione richiesta
- 403 Forbidden: accesso negato
- 404 Not Found: risorsa non trovata
- 500 Internal Server Error: errore del server

package com.esempio.api.controller;

import com.esempio.api.dto.UtenteDTO;
import com.esempio.api.service.UtenteService;
import jakarta.validation.Valid;
import lombok.RequiredArgsConstructor;
import org.springframework.http.HttpStatus;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.*;
import java.net.URI;
import java.util.List;

@RestController
@RequestMapping("/api/utenti")
@RequiredArgsConstructor
public class UtenteController {
    
    private final UtenteService utenteService;
    
    // GET - Recupera tutti gli utenti
    @GetMapping
    public ResponseEntity<List<UtenteDTO>> getAll() {
        return ResponseEntity.ok(utenteService.findAll());
    }
    
    // GET - Recupera un utente per ID
    @GetMapping("/{id}")
    public ResponseEntity<UtenteDTO> getById(@PathVariable Long id) {
        return ResponseEntity.ok(utenteService.findById(id));
    }
    
    // GET - Cerca utenti per nome (query param)
    @GetMapping("/search")
    public ResponseEntity<List<UtenteDTO>> search(@RequestParam String nome) {
        return ResponseEntity.ok(utenteService.findByNome(nome));
    }
    
    // POST - Crea un nuovo utente
    @PostMapping
    @ResponseStatus(HttpStatus.CREATED)
    public ResponseEntity<UtenteDTO> create(@Valid @RequestBody UtenteDTO utente) {
        UtenteDTO saved = utenteService.save(utente);
        return ResponseEntity
            .created(URI.create("/api/utenti/" + saved.getId()))
            .body(saved);
    }
    
    // PUT - Aggiornamento completo
    @PutMapping("/{id}")
    public ResponseEntity<UtenteDTO> update(
            @PathVariable Long id,
            @Valid @RequestBody UtenteDTO utente) {
        return ResponseEntity.ok(utenteService.update(id, utente));
    }
    
    // PATCH - Aggiornamento parziale
    @PatchMapping("/{id}")
    public ResponseEntity<UtenteDTO> patch(
            @PathVariable Long id,
            @RequestBody Map<String, Object> updates) {
        return ResponseEntity.ok(utenteService.patch(id, updates));
    }
    
    // DELETE - Elimina un utente
    @DeleteMapping("/{id}")
    @ResponseStatus(HttpStatus.NO_CONTENT)
    public ResponseEntity<Void> delete(@PathVariable Long id) {
        utenteService.delete(id);
        return ResponseEntity.noContent().build();
    }
}

import org.springframework.data.domain.Page;
import org.springframework.data.domain.PageRequest;
import org.springframework.data.domain.Pageable;
import org.springframework.data.domain.Sort;
import org.springframework.data.web.PageableDefault;

@RestController
@RequestMapping("/api/utenti")
public class UtenteController {
    
    // Paginazione con parametri di default
    @GetMapping
    public ResponseEntity<Page<UtenteResponseDTO>> getAll(
            @PageableDefault(size = 20, sort = "nome", direction = Sort.Direction.ASC) 
            Pageable pageable) {
        return ResponseEntity.ok(utenteService.findAll(pageable));
    }
    
    // Paginazione personalizzata
    @GetMapping("/custom")
    public ResponseEntity<Page<UtenteResponseDTO>> getCustom(
            @RequestParam(defaultValue = "0") int page,
            @RequestParam(defaultValue = "10") int size,
            @RequestParam(defaultValue = "id") String sortBy,
            @RequestParam(defaultValue = "asc") String direction) {
        
        Sort.Direction sortDirection = direction.equalsIgnoreCase("desc") 
            ? Sort.Direction.DESC : Sort.Direction.ASC;
        
        Pageable pageable = PageRequest.of(page, size, Sort.by(sortDirection, sortBy));
        return ResponseEntity.ok(utenteService.findAll(pageable));
    }
    
    // Paginazione con filtri
    @GetMapping("/filter")
    public ResponseEntity<Page<UtenteResponseDTO>> filter(
            @RequestParam(required = false) String nome,
            @RequestParam(required = false) Integer etaMin,
            @RequestParam(required = false) Integer etaMax,
            @PageableDefault(size = 20) Pageable pageable) {
        return ResponseEntity.ok(utenteService.filter(nome, etaMin, etaMax, pageable));
    }
}

Exception Handling

Global Exception Handler con @RestControllerAdvice
package com.esempio.api.exception;

import lombok.AllArgsConstructor;
import lombok.Data;
import org.springframework.http.HttpStatus;
import org.springframework.http.ResponseEntity;
import org.springframework.validation.FieldError;
import org.springframework.web.bind.MethodArgumentNotValidException;
import org.springframework.web.bind.annotation.ExceptionHandler;
import org.springframework.web.bind.annotation.RestControllerAdvice;
import java.time.LocalDateTime;
import java.util.HashMap;
import java.util.Map;

@Data
@AllArgsConstructor
class ErrorResponse {
    private String code;
    private String message;
    private LocalDateTime timestamp;
    private Map<String, String> details;
}

@RestControllerAdvice
public class GlobalExceptionHandler {
    
    // 404 - Risorsa non trovata
    @ExceptionHandler(ResourceNotFoundException.class)
    @ResponseStatus(HttpStatus.NOT_FOUND)
    public ErrorResponse handleNotFound(ResourceNotFoundException ex) {
        return new ErrorResponse(
            "NOT_FOUND",
            ex.getMessage(),
            LocalDateTime.now(),
            new HashMap<>()
        );
    }
    
    // 400 - Validazione fallita
    @ExceptionHandler(MethodArgumentNotValidException.class)
    @ResponseStatus(HttpStatus.BAD_REQUEST)
    public ErrorResponse handleValidation(MethodArgumentNotValidException ex) {
        Map<String, String> errors = new HashMap<>();
        ex.getBindingResult().getAllErrors().forEach(error -> {
            String fieldName = ((FieldError) error).getField();
            String errorMessage = error.getDefaultMessage();
            errors.put(fieldName, errorMessage);
        });
        
        return new ErrorResponse(
            "VALIDATION_ERROR",
            "Errore di validazione",
            LocalDateTime.now(),
            errors
        );
    }
    
    // 400 - Bad request (parametri mancanti, formato errato)
    @ExceptionHandler(MissingServletRequestParameterException.class)
    @ResponseStatus(HttpStatus.BAD_REQUEST)
    public ErrorResponse handleMissingParams(MissingServletRequestParameterException ex) {
        return new ErrorResponse(
            "MISSING_PARAMETER",
            "Parametro mancante: " + ex.getParameterName(),
            LocalDateTime.now(),
            new HashMap<>()
        );
    }
    
    // 409 - Conflitto (email già esistente, ecc.)
    @ExceptionHandler(DuplicateResourceException.class)
    @ResponseStatus(HttpStatus.CONFLICT)
    public ErrorResponse handleDuplicate(DuplicateResourceException ex) {
        return new ErrorResponse(
            "DUPLICATE_RESOURCE",
            ex.getMessage(),
            LocalDateTime.now(),
            new HashMap<>()
        );
    }
    
    // 500 - Errore interno del server
    @ExceptionHandler(Exception.class)
    @ResponseStatus(HttpStatus.INTERNAL_SERVER_ERROR)
    public ErrorResponse handleGeneric(Exception ex) {
        return new ErrorResponse(
            "INTERNAL_ERROR",
            "Errore interno del server",
            LocalDateTime.now(),
            new HashMap<>()
        );
    }
}

// Eccezioni personalizzate

    public ResourceNotFoundException(String message) {
        super(message);
    }
}

class DuplicateResourceException extends RuntimeException {
    public DuplicateResourceException(String message) {
        super(message);
    }
}

@GetMapping("/{id}")
public EntityModel<UtenteDTO> getById(@PathVariable Long id) {
    UtenteDTO utente = service.findById(id);

    resource.add(linkTo(methodOn(UtenteController.class).getById(id)).withSelfRel());

    return resource;
}
